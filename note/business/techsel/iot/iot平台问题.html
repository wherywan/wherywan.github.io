public DefultPropertyMetadata getPropertyMetadata(String deviceId,String identifier)
{
    //缓存取属性对应的json
    String jsonString = redisService.getMetada(deviceId,identifier);
    return new DefultPropertyMetadata(JSON.parseobject(jsonString));
}
